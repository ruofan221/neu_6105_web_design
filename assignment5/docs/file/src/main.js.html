// binary search
{
    function binarysearch(arr, target){
        let hi = arr.length - 1;
        let lo = 0;
        let mid = 0;
        while(lo <= hi){
            /**
             * chop the arr, move the mid to middle of the arr. 
             * using Math.floor to calculate the mid.
             * reference：https://www.w3schools.com/jsref/jsref_floor.asp
             */
            mid = Math.floor((hi + lo) / 2);
            // if arr[mid] > target, move the hi to mid down one
            if(arr[mid] > target) hi = mid - 1;
            // if arr[mid] < target, move the lo to mid up one
            else if(arr[mid] < target) lo = mid + 1;
            // the last condition is arr[mid] equals to target, return mid;
            else return mid;
        };
        // if target were not in the arr, return -1
        return -1;
    };

    if(isFounded === -1) console.log("target number is not in the array " );

    // Bubble sort
    /**
     * repeatedly swapping the adjacent elements if they are in wrong order.
     * so the largest number will sink to the bottom of array
     */
    {
    function bubblesort(arr){
        let N = arr.length - 1;
        for(let i = 0; i <= N; i++){
            for(let j = 0; j <= N - i; j++){
                if(arr[j] > arr[j + 1]) {
                    let temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                };
            };
        };
        return arr;
    }

        function isPalindrome(str){
            /**
             * get the string to charArray
             * reference: https://stackoverflow.com/questions/8761627/string-tochararray-equivalent-on-javascript?lq=1
            */ 
            let charArray = [...str];
            let i = 0;
            let j = charArray.length - 1;
            var isPalindrom = true;
            while(i <= j){
                if(charArray[i] != charArray[j]) isPalindrom = false;
                i++;
                j--;
            }
            return isPalindrom;
        };
    
        /**
         * palindrom test
         */
        console.log();

        let string2 = "abcdedcba";
