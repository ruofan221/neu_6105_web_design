export class Bubblesort {
/**
* @param {number[]} arr - this is the array which is going to be sorted.
* @return {number[]} return sorted array.
*/
bubblesort(arr){
    let N = arr.length - 1;
    for(let i = 0; i <= N; i++){
        for(let j = 0; j <= N - i; j++){
            if(arr[j] > arr[j + 1]) {
                let temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            };
        };
    };
    return arr;
}
}
